30/// Define String.prototype.toAlternatingCase (or a similar function/method
31/// such as to_alternating_case/toAlternatingCase/ToAlternatingCase in your
32/// selected language; see the initial solution for details) such that each
33/// lowercase letter becomes uppercase and each uppercase letter becomes lowercase.
34/// # Example
35/// ``` "hello world".toAlternatingCase() ``` <br>
36/// returns "HELLO WORLD"
37pub fn to_alternating_case(s: &str) -> String {
38    let kata = Kata {
39        level: Level::L8kyu,
40        tags: vec![Tag::Fundamentals],
41        description: String::from("altERnaTIng cAsE <=> ALTerNAtiNG CaSe"),
42    };
43
44    println!(
45        "Level: {:?}, Tags: {:?}, Description: {}",
46        kata.level, kata.tags, kata.description
47    );
48
49    s.chars()
50        .map(|c| {
51            if c.is_uppercase() {
52                c.to_lowercase().to_string()
53            } else if c.is_lowercase() {
54                c.to_uppercase().to_string()
55            } else {
56                c.to_string()
57            }
58        })
59        .collect()
60}

62/// This function takes two numbers as parameters, the first number

83/// Your goal is to return multiplication table for number that is always
84/// an integer from 1 to 10.
85/// # Example
86/// ``` multi_table(5) ``` <br>
87/// returns "1 * 5 = 5\n2 * 5 = 10\n3 * 5 = 15\n4 * 5 = 20\n5 * 5 = 25\n
88/// 6 * 5 = 30\n7 * 5 = 35\n8 * 5 = 40\n9 * 5 = 45\n10 * 5 = 50"
89pub fn multi_table(n: u64) -> String {
90    let kata = Kata {
91        level: Level::L8kyu,
92        tags: vec![Tag::Fundamentals, Tag::Strings],
93        description: String::from("Multiplication table for number"),
94    };
95
96    println!(
97        "Level: {:?}, Tags: {:?}, Description: {}",
98        kata.level, kata.tags, kata.description
99    );
100
101    let mut mult_t = String::from("");
102    for i in 1..10 {
103        let line = format!("{} * {} = {}\n", i, n, i * n);
104        mult_t.push_str(&line);
105    }
106    let last_line = format!("{} * {} = {}", 10, n, 10 * n);
107    mult_t.push_str(&last_line);
108    mult_t
109}
