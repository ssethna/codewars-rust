7/// Your goal is to return multiplication table for number that is always
8/// an integer from 1 to 10.
9/// # Example
10/// ``` multi_table(5) ``` <br>
11/// returns "1 * 5 = 5\n2 * 5 = 10\n3 * 5 = 15\n4 * 5 = 20\n5 * 5 = 25\n
12/// 6 * 5 = 30\n7 * 5 = 35\n8 * 5 = 40\n9 * 5 = 45\n10 * 5 = 50"
13pub fn multi_table(n: u64) -> String {
14    let kata = Kata {
15        level: Level::L8kyu,
16        tags: vec![Tag::Fundamentals, Tag::Strings],
17        description: String::from("Multiplication table for number"),
18    };
19
20    println!(
21        "Level: {:?}, Tags: {:?}, Description: {}",
22        kata.level, kata.tags, kata.description
23    );
24
25    let mut mult_t = String::from("");
26    for i in 1..10 {
27        let line = format!("{} * {} = {}\n", i, n, i * n);
28        mult_t.push_str(&line);
29    }
30    let last_line = format!("{} * {} = {}", 10, n, 10 * n);
31    mult_t.push_str(&last_line);
32    mult_t
33}

35/// Timmy & Sarah think they are in love, but around where they live, they will only
36/// know once they pick a flower each. If one of the flowers has an even number of
37/// petals and the other has an odd number of petals it means they are in love.
38/// Write a function that will take the number of petals of each flower and
39/// return true if they are in love and false if they aren't.
40/// # Example
41/// ``` is_love(5, 7); ``` <br>
42/// returns false
43pub fn is_love(flower1: u16, flower2: u16) -> bool {
44    let kata = Kata {
45        level: Level::L8kyu,
46        tags: vec![Tag::Fundamentals],
47        description: String::from("Opposites Attract"),
48    };
49
50    println!(
51        "Level: {:?}, Tags: {:?}, Description: {}",
52        kata.level, kata.tags, kata.description
53    );
54
55    if (flower1 + flower2) % 2 != 0 {
56        return true;
57    } else {
58        return false;
59    }
60}
