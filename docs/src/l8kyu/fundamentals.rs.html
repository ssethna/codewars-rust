7/// Your task is to find the first element of an array that is not consecutive.
8/// If the whole array is consecutive then return null.
9/// The array will always have at least 2 elements and all elements will be numbers.
10/// The numbers will also all be unique and in ascending order. The numbers could
11/// be positive or negative and the first non-consecutive could be either too!
12/// # Example
13/// ``` [1,2,3,4,6,7,8] ``` <br>
14/// returns 6
15pub fn first_non_consecutive(arr: &Vec<i32>) -> Option<i32> {
16    for i in 1..arr.len() {
17        if arr[i] - arr[i-1] != 1 {
18            return Some(arr[i]);
19        }
20    }
21    None
22}

95/// Define String.prototype.toAlternatingCase (or a similar function/method
96/// such as to_alternating_case/toAlternatingCase/ToAlternatingCase in your
97/// selected language; see the initial solution for details) such that each
98/// lowercase letter becomes uppercase and each uppercase letter becomes lowercase.
99/// # Example
100/// ``` "hello world".toAlternatingCase() ``` <br>
101/// returns "HELLO WORLD"
102pub fn to_alternating_case(s: &str) -> String {
103    let kata = Kata {
104        level: Level::L8kyu,
105        tags: vec![Tag::Fundamentals],
106        description: String::from("altERnaTIng cAsE <=> ALTerNAtiNG CaSe"),
107    };
108
109    println!(
110        "Level: {:?}, Tags: {:?}, Description: {}",
111        kata.level, kata.tags, kata.description
112    );
113
114    s.chars()
115        .map(|c| {
116            if c.is_uppercase() {
117                c.to_lowercase().to_string()
118            } else if c.is_lowercase() {
119                c.to_uppercase().to_string()
120            } else {
121                c.to_string()
122            }
123        })
124        .collect()
125}

148/// Your goal is to return multiplication table for number that is always
149/// an integer from 1 to 10.
150/// # Example
151/// ``` multi_table(5) ``` <br>
152/// returns "1 * 5 = 5\n2 * 5 = 10\n3 * 5 = 15\n4 * 5 = 20\n5 * 5 = 25\n
153/// 6 * 5 = 30\n7 * 5 = 35\n8 * 5 = 40\n9 * 5 = 45\n10 * 5 = 50"
154pub fn multi_table(n: u64) -> String {
155    let kata = Kata {
156        level: Level::L8kyu,
157        tags: vec![Tag::Fundamentals, Tag::Strings],
158        description: String::from("Multiplication table for number"),
159    };
160
161    println!(
162        "Level: {:?}, Tags: {:?}, Description: {}",
163        kata.level, kata.tags, kata.description
164    );
165
166    let mut mult_t = String::from("");
167    for i in 1..10 {
168        let line = format!("{} * {} = {}\n", i, n, i * n);
169        mult_t.push_str(&line);
170    }
171    let last_line = format!("{} * {} = {}", 10, n, 10 * n);
172    mult_t.push_str(&last_line);
173    mult_t
174}
