51/// Your task is to find the first element of an array that is not consecutive.
52/// If the whole array is consecutive then return null.
53/// The array will always have at least 2 elements and all elements will be numbers.
54/// The numbers will also all be unique and in ascending order. The numbers could
55/// be positive or negative and the first non-consecutive could be either too!
56/// # Example
57/// ``` [1,2,3,4,6,7,8] ``` <br>
58/// returns 6
59pub fn first_non_consecutive(arr: &Vec<i32>) -> Option<i32> {
60    let kata = Kata {
61        level: Level::L8kyu,
62        tags: vec![Tag::Arrays, Tag::Fundamentals],
63        description: String::from("Find the first non-consecutive number"),
64    };
65
66    println!(
67        "Level: {:?}, Tags: {:?}, Description: {}",
68        kata.level, kata.tags, kata.description
69    );
70
71    for i in 1..arr.len() {
72        if arr[i] - arr[i - 1] != 1 {
73            return Some(arr[i]);
74        }
75    }
76    None
77}
